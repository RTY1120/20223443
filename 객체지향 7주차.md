## 동적메모리할당

``` csharp

#include <iostream>
using namespace std;

int main()
{
	int size = 50;
	int* array1 = new int[size];	// 크기가 50인 배열 동적 할당

	for (int i = 0; i < 50; i++) {
		array1[i] = 1 + i;			// 배열에 1부터 50까지의 값 저장
	}
	int* array2 = new int[100];		// 크기가 100인 배열 동적 할당

	for (int i = 0; i < 50; i++) {	// array1의 값을 array2의 뒤쪽에 복사
		array2[50 + i] = array1[i];
		cout << array2[50 + i] << endl;
	}
	delete[]array1;					// array1, array2 동적할당 해제
	delete []array2;
}


```

## 템플릿
### 템플릿 함수
```csharp

#include <iostream>
using namespace std;

template <typename T>		// a,b 두 값을 받아 합을 반환하는 sum함수의 템플릿을 정의
T sum(T a, T b) {
	return a + b;
}

int main()
{
	int a = 10, b = 20;
	cout << sum(a, b) << endl;	// 자료형 int로 템플릿 사용

	float a2 = 3.21, b2 = 4.56;	// 자료형 float로 템플릿 사용
	cout << sum(a2, b2) << endl;
}


```
### 템플릿 클래스
```csharp
#include <iostream>
using namespace std;

// 템플릿 클래스 storage 선언
template <typename T> 
class Storage {
	T data;
public:
	Storage() {}
	// 전달받은 값을 멤버변수 data에 저장하는 생성자
	Storage(T data) : data(data) {}
	// 멤버변수 data의 값을 반환
	T Getdata() {
		return data;
	}
	// 멤버변수 data의 값을 설정
	void setData(T data) {
		this->data = data;
	}
};

int main()
{
	Storage <int> A(25);		// 자료형이 int인 Storage클래스를 생성후 멤버변수 data에 25저장
	cout << A.Getdata() << endl;
	A.setData(70);
	cout << A.Getdata() << endl;

	Storage<float> B(1.23);		// 자료형이 float인 Storage클래스를 생성후 멤버변수 data에 1.23저장
	cout << B.Getdata() << endl;
	B.setData(4.56);
	cout << B.Getdata() << endl;
}


```

