## 동적메모리할당

``` csharp

#include <iostream>
using namespace std;

int main()
{
	int size = 50;
	int* array1 = new int[size];	// 크기가 50인 배열 동적 할당

	for (int i = 0; i < 50; i++) {
		array1[i] = 1 + i;			// 배열에 1부터 50까지의 값 저장
	}
	int* array2 = new int[100];		// 크기가 100인 배열 동적 할당

	for (int i = 0; i < 50; i++) {	// array1의 값을 array2의 뒤쪽에 복사
		array2[50 + i] = array1[i];
		cout << array2[50 + i] << endl;
	}
	delete[]array1;					// array1, array2 동적할당 해제
	delete []array2;
}


```
