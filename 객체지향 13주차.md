### 1번
#### 다음과 같이 int형 vector 클래스의 객체를 선언하고 객체에 새로운 데이터를 push_back() 을 여러번 호출하도록 코드를 수정하라. push_back() 함수를 호출할 때 마다 객체에 대해 size와 capacity를 구해서 값의 변화를 분석하라

```csharp
#include <iostream>
#include <vector>
using namespace std;


int main(int argc, char const* argv[])
{
	vector <int> v1;

	for (int i = 0; i < 10; i++) {
		v1.push_back(10-i);
		cout << "size: " << v1.size() << endl;
		cout << "capacity: " << v1.capacity() << endl;
		cout << endl;
	}


```

### 2번
#### 클래스 템플릿 vector를 사용한 프로그램이다. 프로그램 수행 결과를 예측하고 실행 결과의 비교하고 vector 분석하라.

```csharp
#include <vector>
#include <iostream>
using namespace std;
int main()
{
	vector<int> v1, v2, v3;

	v1.push_back(10);
	v1.push_back(20);
	v1.push_back(30);
	v1.push_back(40);
	v1.push_back(50);

	cout << "v1 = ";
	for (auto& v : v1) {
		cout << v << " ";
	}
	cout << endl;

	// v1의 시작부터 끝까지의 원소를 v2에 초기화
	v2.assign(v1.begin(), v1.end());
	cout << "v2 = ";
	for (auto& v : v2) {
		cout << v << " ";
	}
	cout << endl;

	// 기존 원소를 다 지우고 3개의 저장공간에 6을 대입
	v3.assign(3, 6);
	cout << "v3 = ";
	for (auto& v : v3) {
		cout << v << " ";
	}
	cout << endl;

	// 기존 원소를 다 지우고 5, 6, 7을 대입
	v3.assign({ 5, 6, 7 });
	for (auto& v : v3) {
		cout << v << " ";
	}
	cout << endl;

	// i는 v1의 첫번재 원소를 참조함
	int& i = v1.at(0);

	cout << "v1 첫 번째 원소의 값:  " << i << endl;

	if (v1 == v2) cout << "v1과 v2는 같다." << endl;
	else cout << "v1과 v2는 다르다" << endl;

	i = 80;
	const int& j = v1.at(0);

	cout << "값을 변경 후 v1 첫 번째 원소의 값:  " << j << endl;

	// v1의 첫번째 원소는 80, v2의 첫번째 원소는 10
	if (v1 == v2) cout << "v1과 v2는 같다." << endl;
	else cout << "v1과 v2는 다르다" << endl;
}
```

### 3번
#### 사각형을 표현하는 클래스를 정의하고 클래스를 vector 를 이용하여 저쟝하고 처리하는 프로그램을 작성하라.

```csharp
#include <vector>
#include <iostream>
using namespace std;

class Rectangle {
	int height;
	int base;
public:
	Rectangle(int h, int b) : height(h), base(b) {};

	void printInfo() {
		cout << "높이: " << height << ", " << "밑변: " << base << endl;
	}

	int getArea() {
		return height * base;
	}
};


int main()
{
	vector<Rectangle> v;
	v = { {10, 20},{30, 40},{50, 60},{4, 5},{100, 7} };
	v[1].printInfo();
	v.push_back({ 100,750 });
	cout << v[5].getArea() << endl;
}
```

### 7번
#### 아래의 링크를 참조하여 vector 클래스의 다른 멤버 함수와 연산자를 활용한 프로그램을 작성하라.

```csharp
#include <vector>
#include <iostream>
using namespace std;



int main()
{
	vector<int> v1 = { 1, 2, 3, 4 };
	vector<int> v2 = { 1, 2, 3, 4 };
	vector<int> v3 = { 2, 4, 6 };
	vector<int> v4 = { 1, 2, 3 };


	// operator!=, ==
	if (v1 != v3) {
		cout << "다름" << endl;
	}
	else {
		cout << "같음" << endl;
	}

	if (v1 == v2) {
		cout << "같다" << endl;
	}
	else {
		cout << "다름" << endl;
	}
	// operator< , <=, >, >= 
	if (v1 < v3) {
		cout << "v3가 v1보다 크다" << endl;
	}
	else {
		cout << "v3가 v1보다 같거나 작다" << endl;
	}

	if (v1 <= v4) {
		cout << "v4가 v1보다 같거나 크다" << endl;
	}
	else {
		cout << "v4이 v1보다 작다" << endl;
	}
}
```
