#다형성
## 가상함수
``` csharp
#include <iostream>
using namespace std;

class Animal {
public:
 //   void speak() {       가상함수를 사용하지 않았기 때문에 정적으로 동작
    virtual void speak(){       // 가상함수 virtual을 사용시 정상적으로 출력됨
        cout << "동물이 소리를 냅니다." << endl;
    }
};

class Dog : public Animal {
public:
    void speak() override {
        cout << "멍멍!" << endl;
    }
};

class Cat : public Animal {
public:
    void speak() override {
        cout << "야옹!" << endl;
    }
};

int main() {

    Animal a1;
    Dog a2;
    Cat a3;

    a1.speak();
    a2.speak();
    a3.speak();

    return 0;
}

```
