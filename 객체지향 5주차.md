        cout << "멍멍!" << endl;

    Cat a3;

		return y;

		this->y = y;

### 2번. 클래스 Date와 관련한 멤버함수를 구현하시오
```csharp
#include <iostream>
using namespace std;

class Date {
	int year;
	int month;
	int day;
public:
	Date() {};
	Date(int y, int m, int d) {
		year = y, month = m, day = d;
	}
	~Date() {};
	void setYear(int y) { year = y; }
	void setMonth(int m) { month = m; }
	void setDay(int d) { day = d; }
	int getYear() {
		return year;
	}
	int getMonth() {
		return month;
	}
	int getDay() {
		return day;
	}
	void printDate() {
		cout << year << "년 " << month << "월 " << day << "일" << endl;
	}
};

int main(int argc, char const* argv[])
{
	Date A;
	Date B(2025, 10, 12);

	A.setYear(2003);
	A.setMonth(11);
	A.setDay(20);

	A.printDate();
	B.printDate();
}


```
