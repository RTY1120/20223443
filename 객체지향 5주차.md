### 2. 배열의 크기는 100으로 고정
####
front():첫 번째 원소에 접근 (반환)
back(): 마지막 원소에 접근 (반환)
size(): 현재 원소 개수 반환
empty(): 배열이 비어 있는지 여부를 boolean(true/false)으로 반환
capacity(): 배열 클래스에 할당된 메모리 공간의 크기 반환
at(int idx): idx에 있는 값 반환
insert(int idx, data): idx에 값 저장
push_back( data): 배열의 끝에 원소를 추가
pop_back(); 배열의 마지막 원소를 제거

``` csharp

#include <iostream>
using namespace std;

class MyArray {
	int buf[100];
	bool checkEmpty[100];	// 배열이 비어있는지 확인용
	int lastNum;
public:
	MyArray() {
		lastNum = 0;
		for (int i = 0; i < 100; i++) {		//배열을 모두 true로 초기화
			checkEmpty[i] = true;
		}
	}
	void setNum(int a, int b) {
		// a번째 인덱스에 b를 대입
		buf[a] = b;			

		// a번째 인덱스가 lastNum의 값보다 클 경우 lastNum에 b의 값 대입
		if (a >= lastNum) {	
			lastNum = a;
		}

		// buf의 a번째 인덱스에 값을 넣을 경우 checkEmpty[a]를 false로 변경
		checkEmpty[a] = false;
	}
	int getNum(int a) {
		return buf[a];
	}
	int front() {
		return buf[0];
	}
	int back() {
		return buf[lastNum];
	}
	bool empty(int a) {
		return checkEmpty[a];
	}
};

int main() {
	MyArray a;
	a.setNum(0, 10);
	a.setNum(1, 30);
	a.setNum(3, 5);
	
	cout << a.front() << endl;
	cout << a.getNum(0) << endl;
	cout << a.back() << endl;
	cout << a.empty(1) << endl;
	cout << a.empty(2) << endl;
}

```
